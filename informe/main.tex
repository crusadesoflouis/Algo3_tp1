  \Procedure{CantLibres}{\texttt{vector<int>} elementos,\texttt{vector<bool>} validar} $\to $ \texttt{int}
    \State \textbf{//n es el tamaño de elementos.}
    \State fin $\gets$ elementos.size();
    \For {$i = 0; i < n; i{+}{+}$}
    \State j $\gets$ i;
    \State \While{ (j $<$ fin) }
    \State \texttt{pila<int>} pilaRoja
    \State \texttt{pila<int>} pilaAzul
    \State Rojos(elementos,validar,pilaRoja,i,j)
    \State EstablecerValidos(validar,pilaroja)
    \State Azul(elementos,validar,pilaAzul,i,j)
    \State EstablecerValidos(validar,pilaroja)
	\State	res $\gets$ maximo(pilaAzul,PilaRoja)
	\EndWhile
	\EndFor
\State return res
\end{algorithmic}
